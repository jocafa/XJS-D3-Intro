API

Scales


var scale = d3.scale.linear()
  .domain([0, 1])
  .range([0, 100]);

scale(0.5);

Selections


d3.selectAll('#foo .bar')
  .style('opacity', 0.5)
  .style('color', 'steelblue')
  .text(function () {
    return '[' + this.innerText + ']';
  });

Joins


var listItems = d3.selectAll('#staff li')
  .data([ 'Philip J. Fry', 'Prof. Hubert J. Farnsworth', …]);

listItems.enter()
  .append('li');

listItems
  .text(function (d) { return d; });

listItems.exit()
  .remove();

Transitions


listItems.enter()
  .append('li')
  .style('opacity', 0);

listItems
  .text(function (d) { return d; })
  .transition()
  .style('opacity', 1);

listItems.exit().transition()
  .style('opacity', 0);
  .remove();
